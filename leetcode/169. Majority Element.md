hash table:
```
class Solution {
public:
    int majorityElement(vector<int>& nums) {
        const int size = nums.size();
        unordered_map<int, int> map;
        for (int i = 0; i < size; ++i)
            if (++map[nums[i]] > size/2)
                return nums[i];
    }
};
```
