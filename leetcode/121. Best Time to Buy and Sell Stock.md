class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int sz = prices.size();
        if (sz <= 1) return 0;
        int profit = 0, min = prices[0];
        for (int i = 1; i < sz; ++i) {
            min = (prices[i] < min) ? prices[i] : min;
            profit = (prices[i] - min > profit) ? (prices[i] - min) : profit;
        }
        return profit;
    }
};
