/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
int* twoSum(int* nums, int numsSize, int target) {
    int *num = (int *) malloc (sizeof(int) * 2), a, b;
    for (int i = 0; i < numsSize - 1; i++) {
        a = nums[i];
        for (int j = i + 1; j < numsSize; j++) {
            b = nums[j];
            if (target - a == b) {
                num[0] = i;
                num[1] = j;
                return num;
            }
        }
    }
}
