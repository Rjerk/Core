/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* deleteDuplicates(ListNode* head) {
        auto n = head;
        while (n != nullptr && n->next != nullptr) {
            if (n->val == n->next->val)
                n->next = n->next->next;
            else
                n = n->next;
        }
        return head;
    }
};
