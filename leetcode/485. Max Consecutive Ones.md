class Solution {
public:
    int findMaxConsecutiveOnes(vector<int>& nums) {
        int max = 0, count = 0, sz = nums.size();
        for (int i = 0; i < sz; ++i) {
            if (nums[i] == 1) {
                ++count;
                if (max < count)
                    max = count;
            } else {
                if (max < count)
                    max = count;
                count = 0;
            }
        }
        return max;
    }
};
