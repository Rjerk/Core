class Solution {
public:
    vector<int> plusOne(vector<int>& digits) {
        int sz = digits.size();
        vector<int> vd;
        digits[sz-1] += 1;
        if (digits[sz-1] < 10) return digits;
        for (int i = sz - 1; i >= 0; --i) {
            if (digits[i] > 9) {
                digits[i] %= 10;
                vd.push_back(digits[i]);
                if (i == 0) {
                    vd.push_back(1);
                } else {
                    digits[i-1] += 1;
                }
            } else {
                vd.push_back(digits[i]);
            }
        }
        std::reverse(begin(vd), end(vd));
        return vd;
    }
};

class Solution {
public:
    vector<int> plusOne(vector<int>& digits) {
        int n = digits.size();
        for (int i = n - 1; i >= 0; --i) {
            if (digits[i] == 9) {
                digits[i] = 0;
            } else {
                digits[i]++;
                return digits;
            }
        }
        digits[0] = 1;
        digits.push_back(0);
        return digits;
    }
};
