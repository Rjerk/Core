class Solution {
public:
    int wiggleMaxLength(vector<int>& nums) {
        int f = 1;
        int d = 1;
        int sz = nums.size();
        for (int i = 1; i < sz; ++i) {
            if (nums[i] > nums[i-1]) f = d + 1;
            else if (nums[i] < nums[i-1]) d = f + 1;
        }
        return min(sz, max(f, d));
    }
};
