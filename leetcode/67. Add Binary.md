class Solution {
public:
    string addBinary(string a, string b) {
        int index_a = a.size()-1;
        int index_b = b.size()-1;
        string newstr = "";
        int carry = 0;
        while (index_a >= 0 || index_b >= 0 || carry > 0) {
            carry += (index_a >= 0 ? a[index_a--] - '0' : 0);
            carry += (index_b >= 0 ? b[index_b--] - '0' : 0);
            newstr = char(carry % 2 + '0') + newstr;
            carry /= 2;
        }
        return newstr;
    }
};
