class Solution {
public:
    int missingNumber(vector<int>& nums) {
        int sz = nums.size();
        vector<int> vec(sz+1, -1);
        for (int i = 0; i < sz; ++i) {
            vec[nums[i]] = nums[i];
        }
        for (int i = 0; i < sz+1; ++i)
            if (vec[i] < 0)
                return i;
    }
};
