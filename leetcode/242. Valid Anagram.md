class Solution {
public:
    bool isAnagram(string s, string t) {
        if (s.length() != t.length()) return false;
        std::unordered_map<char, int> m1;
        for (auto c : s) m1[c] += 1;
        for (auto c : t) m1[c] -= 1;
        for (auto s : m1)
            if (s.second)
                return false;
        return true;
    }
};
