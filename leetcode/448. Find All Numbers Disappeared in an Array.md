class Solution {
public:
    vector<int> findDisappearedNumbers(vector<int>& nums) {
        vector<int> s;
        int sz = nums.size();
        for (int i = 0; i < sz; ++i) {
            int index = abs(nums[i]) - 1;
            if (nums[index] > 0) nums[index] = -nums[index];
        }
        for (int i = 0; i < sz; ++i) {
            if (nums[i] > 0)
                s.push_back(i+1);
        }
        return s;
    }
};
