# 206. Reverse Linked List

## Question

> Reverse a singly linked list.

> C++
```
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        //
    }
};
```

## Solution

### 思路1

使用三个指针从头到尾来reverse。

9ms:

```
class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        if (head == NULL || head->next == NULL)
            return head;
        ListNode *p = head, *q = head->next;
        while (q != NULL) {
            ListNode* r = q->next;
            if (p == head) {
                p->next = NULL;
            }
            q->next = p;
            p = q;
            q = r;
        }
        head = p;
        return head;
    }
};
```

#### 改进

反正head再也用不到，直接用head来迭代好了。

9ms:

```
class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        if (head == NULL || head->next == NULL)
            return head;
        ListNode *q = head->next;
        head->next = NULL;
        while (q != NULL) {
            ListNode* r = q->next;
            q->next = head;
            head = q;
            q = r;
        }
        return head;
    }
};
```

### 思路2

递归解法。

9ms:

```
class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        if (head == NULL || head->next == NULL)
            return head;
        ListNode* p = head->next;
        head->next = NULL;
        ListNode* newhead = reverseList(p);
        p->next = head;
        head = NULL;
        return newhead;
    }
};
```
