class Solution {
public:
    int climbStairs(int n) {
        if (n == 0 || n == 1) return 1;
        if (n == 2) return 2;
        int i = 3;
        int pre1 = 1, pre2 = 2;
        int result_n = 0;
        while (i <= n) {
            result_n = pre1 +pre2;
            pre1 = pre2;
            pre2 = result_n;
            ++i;
        }
        return result_n;
    }
};
