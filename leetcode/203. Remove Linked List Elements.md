/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* removeElements(ListNode* head, int val) {
        if (head == nullptr) return head;
        auto n = head;
        while (n->next != nullptr) {
            if (n->next->val != val) {
                n = n->next;
            } else {
                n->next = n->next->next;
            }
        }
        if (head->val == val) head = head->next;
        return head;
    }
};
