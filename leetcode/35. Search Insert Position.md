class Solution {
public:
    int searchInsert(vector<int>& nums, int target) {
        vector<int>::size_type i;
        for (i = 0; i < nums.size(); ++i)
            if (target <= nums[i])
                return i;
        return i;
    }
};

class Solution {
public:
    int searchInsert(vector<int>& nums, int target) {
        vector<int>::size_type low = 0, high = nums.size()-1;
        while (low < high) {
            vector<int>::size_type mid = low + (high - low) / 2;
            if (target >= nums[mid])
                low = mid;
            else
                high = mid;
        }
        return low;
    }
};
